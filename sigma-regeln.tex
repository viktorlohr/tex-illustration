\documentclass{standalone}
\usepackage{pgfplots}
\usepgfplotslibrary{fillbetween}
%\usetikzlibrary{patterns.meta}
\usetikzlibrary{patterns.meta}
\pgfplotsset{compat=1.18}
\renewcommand{\familydefault}{\sfdefault}

\definecolor{myDarkGray}{gray}{0.20}  
\definecolor{myMidGray}{gray}{0.55}   
\definecolor{myLightGray}{gray}{0.75}  

\begin{document}
\begin{tikzpicture}
\begin{axis}[
    legend style = {
        at={(0.95,0.95)}, 
        anchor=north east,
        cells={anchor=west},
        font=\huge
    },
    legend image post style={scale=1.5},
    axis lines = left,
    samples=100,
    domain=40:160,
    xtick={55, 70, 85, 100, 115, 130, 145},
    yticklabels={},
    scaled ticks=false,
    xticklabels={
        $\mu{\!}-{\!}3\sigma$,
        $\mu{\!}-{\!}2\sigma$, 
        $\mu{\!}-{\!}\sigma$, 
        $\mu$, 
        $\mu{\!}+{\!}\sigma$, 
        $\mu{\!}+{\!}2\sigma$,
        $\mu{\!}+{\!}3\sigma$
    },
    scaled y ticks=false,
    y tick label style={
        /pgf/number format/fixed,
        /pgf/number format/precision=2,
    },
    height=6cm,
    width=12cm,
    no markers,
    every axis plot/.append style={thick},
    legend style = {
        nodes={scale=0.5}
    }
]   
    % Gaußkurve
    \addplot [black, name path=gauss, forget plot] 
        {1/(15*sqrt(2*pi)) * exp(-((x-100)^2)/(500))};

    % Vertikale Linien und Pfeile
    \draw [black, thick, dotted] (axis cs:100, 0) -- (axis cs:100,0.3);
    \pgfplotsinvokeforeach{55,145}{
        \draw [myMidGray, thick, dashed] (axis cs:#1, 0) -- (axis cs:#1,0.0075);
        \draw [myMidGray, thick, -stealth] (axis cs: 100,0.0075) 
            -- node[above, near end] {$3\sigma$}
            (axis cs:#1,0.0075);
    }
    \pgfplotsinvokeforeach{70,130}{
        \draw [myLightGray, thick, dashed] (axis cs:#1, 0) -- (axis cs:#1,0.015);
         \draw [myLightGray, thick, -stealth] (axis cs: 100,0.015) 
            -- node[above, near end] {$2\sigma$}
            (axis cs:#1,0.015);
    }
    \pgfplotsinvokeforeach{85,115}{
        \draw [myDarkGray, thick, dashed] (axis cs:#1, 0) -- (axis cs:#1,0.0225);
        \draw [myDarkGray, thick, -stealth] (axis cs: 100,0.0225) 
            -- node[above, near end] {$\sigma$}
            (axis cs:#1,0.0225);
    }

    % Unsichtbarer Pfad auf x-Achse für Flächenberechnung
    \path[name path=axis] (axis cs:40,0) -- (axis cs:160,0);

    % Flächen
    \addplot [
        pattern={Dots[distance=5pt]}, 
        pattern color=myMidGray,
        area legend
        ] 
        fill between[
        of=gauss and axis, 
        soft clip={domain=55:145}
        ];
        \addlegendentry{\color{myMidGray} $99,7\%$}
    \addplot [
        pattern={Lines[angle=135, distance=5pt]}, 
        pattern color=myLightGray,
        thick,
        area legend
    ] 
    fill between[
        of=gauss and axis, 
        soft clip={domain=70:130}
        ];
    \addlegendentry{\color{myLightGray} $95,4\%$}
    \addplot [
        pattern={Lines[angle=45]}, 
        pattern color=myDarkGray,
        area legend
        ] 
    fill between[
        of=gauss and axis, 
        soft clip={domain=85:115}
        ];
    \addlegendentry{\color{myDarkGray} $68,3\%$}
    

%TODO: Legende hübscher machen

    
\end{axis}
\end{tikzpicture}
\end{document}
