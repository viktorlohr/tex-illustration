\documentclass{standalone}
\usepackage{pgfplots}
\usetikzlibrary{arrows.meta}
\pgfplotsset{compat=1.18}
\renewcommand{\familydefault}{\sfdefault}

\definecolor{myDarkGray}{gray}{0.3}  
\definecolor{myMidGray}{gray}{0.55}   
\definecolor{myLightGray}{gray}{0.75}  

\begin{document}
\begin{tikzpicture}
\begin{axis}[
    view={20}{20},
    axis lines=center,
    xlabel=$x$, zlabel=$y$,
    colormap={customgray}{
    color(0cm)=(myLightGray); 
    color(1cm)=(myMidGray)
    },
    xtick=\empty,
    ytick=\empty,
    ztick=\empty
]
    % Rotationsfläche
    \addplot3[
        surf,
        opacity=0.1,
        fill opacity=0.3,
        z buffer=sort,
        samples=20,
        domain=0.5:1.5,
        y domain=0:2*pi,
        variable=\x,
        variable y=\y
    ] (
        {\x},
        {sqrt(\x) * cos(deg(\y))},
        {sqrt(\x) * sin(deg(\y))}
    );

    % Blaue Kurve
    \addplot3[
        domain=0:2,
        variable=\x,
        samples=50,
        samples y=0,
        very thick,
        myMidGray
    ] (
        {\x},
        {0},
        {sqrt(\x)}
    )
    node[above, pos=0.5] {$G_f$};

    %rote Kurve
    \addplot3[
        domain=0.5:1.5,
        variable=\x,
        samples=50,
        samples y=0,
        very thick,
        myDarkGray
    ] (
        {\x},
        {0},
        {sqrt(\x)}
    );

    % Punkte a und b
    \addplot3[
        myDarkGray,
        mark=*,
        mark size=2pt
    ] coordinates {(0.5,0,0)}
    node[below] {$a$};

    \addplot3[
        myDarkGray,
        mark=*,
        mark size=2pt
    ] coordinates {(1.5,0,0)}
    node[below] {$b$};

    %Projektionen
    \addplot3[
        dashed,
        myDarkGray
    ] coordinates {(0.5,0,0) (0.5,0,{sqrt(0.5)})};

    \addplot3[
        dashed,
        myDarkGray
    ] coordinates {(1.5,0,0) (1.5,0,{sqrt(1.5)})};

    % Rotationspfeil
    \addplot3[
        very thick, -{Stealth[reversed, length=3mm, width=2mm]},
        shorten >=-1pt,
        line cap=round,
        variable y=\y,
        y domain =0.55*pi:pi,
        myDarkGray
    ] ({1},
        {sqrt(1) * cos(deg(\y))},
        {sqrt(1) * sin(deg(\y))}
    );


% TODO: DRY code


\end{axis}
\end{tikzpicture}
\end{document}
